fn term(input: &[u8]) -> IResult<&[u8], i64> {
  let (input, init) = factor(input)?;
  fold_many0(
    pair(one_of("*/"), factor),
    move || init,
    |acc, (op, val)| {
      if op == '*' {
        acc * val
      } else {
        acc / val
      }
    },
  )(input)
}
fn term_test() {
  assert_eq!(term(" 12 *2 /  3"), Ok(("", 8)));
  assert_eq!(term(" 2* 3  *2 *2 /  3"), Ok(("", 8)));
  assert_eq!(term(" 48 /  3/2"), Ok(("", 8)));
}
